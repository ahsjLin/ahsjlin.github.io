多個figure顯示
import matplotlib.pyplot as plt
import numpy as np

x = np.linspace(-3,3, 50)

y1 = 2*x + 1

y2 = x**2

##
# plt.figure(num=3) -> Meaning figure order is third.
# plt.figure(figsize=(8,5)) -> figure scale
##


# First Figure
plt.figure()
plt.plot(x, y1)

# Second Figure
plt.figure()
plt.plot(x, y2)

# display figure
plt.show()

# 如果要在同一條線上作圖, 只要figure就行了。


plt.plot()的參數 
# line color
plt.plot(color='red')

# line width
plt.plot(linewidth=1.0)

# line style
plt.plot(linestyle='--')
