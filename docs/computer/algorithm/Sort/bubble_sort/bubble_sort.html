Bubble Sort

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
using namespace std;

void Swap(char *a, char *b) {
	int lenA = strlen(a);
	int lenB = strlen(b);

	char *tmp = (char*)calloc((lenA+1), sizeof(char));

	strncpy(tmp, a, lenA+1);

	strncpy(a, b, lenB+1);
	a[lenB] = '\0';

	strncpy(b, tmp, lenA+1);
	b[lenA] = '\0';

	free(tmp);
}

void sort(char **array, int num) {
    for (int i = 0; i < num; i++) {
        for (int j = 0; j < num-1; j++) {
            if ((int)array[j][0] > (int)array[j+1][0]) {
		    Swap(array[j], array[j+1]);
            }
        }
    }
}

int main() {
    int num, len;
    printf("Input number and length:\n");
    scanf("%d %d", &num, &len);
    
    char **array = (char **) malloc((sizeof(char *)) * num);
    for (int i = 0; i < num; i++) {
        array[i] = (char *) malloc(sizeof(char) * len);
    }
    
    for (int i = 0; i < num; i++) {
		scanf("%s", array[i]);
    }

    puts("=============");
    
    // sort the string
    sort(array, num);
    
    // print the string
    for (int i = 0; i < num; i++) {
            printf("%s\n", array[i]);
    }
    
    free(array);
    return 0;
}
